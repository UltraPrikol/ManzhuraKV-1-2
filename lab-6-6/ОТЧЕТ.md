struct Product {
    name: String,
    price: f64,
    available: bool,

    amount: f64,

// --- Задание 1: Анализ продуктов ---
fn analyze_products(products: &[Product]) -> (f64, usize, Vec<&Product>) {
    let count = products.len();
    
    // Средняя цена (если список пуст, вернем 0.0)
    let avg_price = if count == 0 {
        0.0
    } else {
        products.iter().map(|p| p.price).sum::<f64>() / count as f64
    };

    // Количество доступных
    let available_count = products.iter().filter(|p| p.available).count();

    // Список дорогих продуктов (> 100)
    let expensive_products: Vec<&Product> = products
        .iter()
        .filter(|p| p.price > 100.0)
        .collect();

    (avg_price, available_count, expensive_products)
}

// --- Задание 2: Валидация цепочки заказов ---
fn validate_orders(orders: &[Order]) -> Result<Vec<&Order>, OrderError> {
    // В Rust метод collect может превратить Iterator<Result<T, E>> в Result<Vec<T>, E>
    // Он вернет первую встреченную ошибку или вектор всех успешных элементов.
    orders.iter()
        .map(|o| {
            if o.amount > 0.0 {
                Ok(o)
            } else {
                Err(OrderError::InvalidAmount(o.id))
            }
        })
        .collect()
}

// --- Задание 3: Итератор Фибоначчи ---
struct Fibonacci {
    current: u64,
    next: u64,
}

impl Fibonacci {
    fn new() -> Self {
        Fibonacci { current: 0, next: 1 }
    }
}

impl Iterator for Fibonacci {
    type Item = u64;
    
    fn next(&mut self) -> Option<Self::Item> {
        let current_val = self.current;
        
        // Математическая логика: F(n) = F(n-1) + F(n-2)
        let next_val = self.current.checked_add(self.next);
        
        match next_val {
            Some(sum) => {
                self.current = self.next;
                self.next = sum;
                Some(current_val)
            }
            None => None, // Защита от переполнения u64
        }
    }
}

**Контрольные вопросы**
1.  Какой язык оказался наиболее выразительным для ФП и почему?
Наиболее выразительным для ФП является Scala. Она изначально проектировалась как гибрид объектно-ориентированного и функционального подходов. 
В ней наиболее развита система типов (высшего порядка), мощный механизм неявных преобразований и встроенная поддержка монад («контейнеров»), что позволяет писать очень лаконичный и декларативный код.

2.  Какие компромиссы между безопасностью типов и продуктивностью разработки вы заметили?
Python обеспечивает высочайшую скорость написания кода (продуктивность), но платит за это риском ошибок в рантайме. 
Rust требует гораздо больше времени на начальном этапе (борьба с компилятором), но гарантирует отсутствие целых классов ошибок памяти. 
Scala находится посередине: строгая типизация помогает ловить ошибки, но сложность некоторых абстракций может замедлять разработку.

3.  Как разные языки решают проблему побочных эффектов?
Python: Никак не ограничивает побочные эффекты на уровне языка, полагаясь на дисциплину программиста.

Scala: Использует чистые функции и оборачивает эффекты в специальные типы (Option, Try, Future), делая их явными.

Rust: Контролирует побочные эффекты через систему владения и заимствования. Вы не можете случайно изменить данные, если у вас нет на это исключительного права (&mut).

4.  Какой язык вы бы выбрали для high-performance приложения и почему?
Для максимально высокой производительности однозначным выбором является Rust. 
Он компилируется в машинный код (как C++), не имеет сборщика мусора (GC), который вызывает паузы в работе, и предоставляет «абстракции с нулевой стоимостью». 
Это позволяет коду работать на максимально возможной скорости железа.

5.  Какие особенности Rust делают его уникальным среди других языков?
Главная особенность Rust — это Borrow Checker (проверка заимствований). 
Это уникальная система, которая обеспечивает полную безопасность памяти и защиту от «состояний гонки» (data races) во время компиляции без использования сборщика мусора. 
Это первый язык, решивший проблему безопасного системного программирования в промышленном масштабе.
