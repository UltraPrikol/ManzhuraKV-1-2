    if (!input) return '';

        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;')
        .replace(/'/g, '&#x27;')
        .replace(/\//g, '&#x2F;');
};

app.get('/', (req, res) => {

    });
});

app.post('/comment', (req, res) => {

    username = sanitizeHtml(username || 'Anonymous');
    comment = sanitizeHtml(comment || '');
    
    // БЕЗОПАСНО: Параметризованный запрос
    db.run(`INSERT INTO comments (username, comment) VALUES (?, ?)`,

### Контрольные вопросы
- Какие CVE были обнаружены в зависимостях? Каковы их последствия?
    В установленных нами устаревших версиях библиотек сканер зафиксировал несколько классических уязвимостей (CVE — Common Vulnerabilities and Exposures):
    ejs@2.6.1 (CVE-2019-10744 / CVE-2022-29078): * Уязвимость: RCE (Remote Code Execution) через загрязнение прототипа (Prototype Pollution) или некорректную санитизацию параметров шаблонизатора.
        Последствия: Атакующий может передать злонамеренный код в параметры рендеринга, что позволит ему удаленно выполнять любые системные команды на сервере, где запущен Node.js. Это критический уровень опасности (Critical).
    axios@0.18.0 (CVE-2020-28168):
        Уязвимость: SSRF (Server-Side Request Forgery).
        Последствия: Из-за некорректной обработки относительных и абсолютных путей в старой версии Axios, атакующий может заставить твой сервер отправлять HTTP-запросы на скрытые внутренние ресурсы (например, к метаданным облачного провайдера AWS/GCP по адресу 169.254.169.254 или к локальным базам данных, закрытым снаружи файрволом).
    express@4.16.0 и body-parser@1.18.3 (CVE-2019-10744 в зависимом модуле qs):
        Уязвимость: Prototype Pollution / Denial of Service (DoS).
        Последствия: Передавая специально сформированные вложенные JSON-объекты, злоумышленник может перезаписать встроенные свойства базовых объектов JavaScript. Это приводит либо к падению сервера (отказ в обслуживании), либо к изменению логики авторизации приложения.
- В чем разница между `npm audit` и `snyk test`?
    npm audit: Проверяет только Node.js/JS, использует стандартную базу GitHub, бесплатен.
    snyk test: Мультиязычен (JS, Python, Docker, IaC), использует собственную более глубокую базу уязвимостей, легче внедряется в CI/CD.
- Как OWASP ZAP обнаруживает XSS-уязвимости? Объясните механизм.
    Работает как DAST (черный ящик):
    Находит все формы и параметры.
    Отправляет в них вредоносный JS-код (фаззинг).
    Проверяет ответ сервера: если код вернулся на страницу в исходном виде (без экранирования символов < и >), ZAP фиксирует XSS.
- Почему CSP (Content Security Policy) эффективен против XSS, даже если разработчик забыл экранировать вывод?
    CSP — это защита на стороне браузера. Даже если злоумышленник смог внедрить свой скрипт в HTML-код, политика CSP запрещает выполнять встроенный код (без директивы 'unsafe-inline') и загружать скрипты со сторонних сайтов. Браузер его просто блокирует.
- Какие ограничения есть у DAST-инструментов по сравнению с SAST?
    Не видит код: не найдет зашитые в кодовой базе пароли и ключи.
    Зависимость от запуска: требует полностью рабочего, развернутого приложения.
    Слепые зоны: может пропустить эндпоинты, на которые нет ссылок в интерфейсе.
    Размытый фикс: пишет, где уязвимость на сайте, но не указывает конкретную строку кода, которую нужно исправить.
