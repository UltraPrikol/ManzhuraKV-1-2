```python
from functools import reduce, wraps

# =========================
# ЗАДАНИЕ 1
# Анализ данных студентов
# =========================

def analyze_students(students):
    """Анализ данных студентов"""
    total_students = len(students)

    grades = list(map(lambda s: s["grade"], students))

    average_grade = (
        reduce(lambda a, b: a + b, grades, 0) / total_students
        if total_students > 0 else 0
    )

    top_students = list(
        map(
            lambda s: s["name"],
            filter(lambda s: s["grade"] >= 90, students)
        )
    )

    return {
        "average_grade": average_grade,
        "top_students": top_students,
        "total_students": total_students
    }


# =========================
# ЗАДАНИЕ 2
# Декоратор логирования
# =========================

def logger(func):
    """Декоратор для логирования"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        print(f"[LOG] Функция: {func.__name__}")
        print(f"[LOG] Аргументы: args={args}, kwargs={kwargs}")
        result = func(*args, **kwargs)
        print(f"[LOG] Результат: {result}")
        return result
    return wrapper


@logger
def greet(name):
    return f"Привет, {name}!"


# =========================
# ЗАДАНИЕ 3
# Генератор простых чисел
# =========================

def prime_generator():
    """Генератор простых чисел"""
    def is_prime(n):
        if n < 2:
            return False
        for i in range(2, int(n ** 0.5) + 1):
            if n % i == 0:
                return False
        return True

    number = 2
    while True:
        if is_prime(number):
            yield number
        number += 1

def main():

    stats = analyze_students(students)

    primes = prime_generator()

        print(next(primes), end=" ")
    print()


if __name__ == "__main__":
```

**Контрольные вопросы**
1.  Что такое функции первого класса и как Python их поддерживает?
Функции первого класса - это функции, которые в Python рассматриваются как обычные объекты. 
Их можно присваивать переменным, передавать в качестве аргументов другим функциям, возвращать из функций и хранить в структурах данных.

2.  В чем разница между lambda-функциями и обычными функциями?
Lambda-функции отличаются от обычных функций тем, что они являются анонимными, состоят из одного выражения и автоматически 
возвращают результат этого выражения. Обычные функции создаются с помощью def, могут содержать несколько выражений и инструкции управления.

3.  Как работают замыкания и для чего они используются?
Замыкания — это функции, которые сохраняют доступ к переменным из внешней области видимости даже после завершения работы внешней функции. 
Они используются для хранения состояния, инкапсуляции данных и создания функций с настраиваемым поведением.

4.  Что такое декораторы и как создавать декораторы с параметрами?
Декораторы — это специальные функции, которые позволяют изменять или расширять поведение других функций без изменения их исходного кода. 
Декораторы с параметрами создаются с помощью вложенных функций, где внешняя функция принимает параметры, а внутренняя — декорируемую функцию.

5.  Какие преимущества дают генераторы по сравнению со списками?
Генераторы обладают преимуществами по памяти и производительности, так как создают значения по мере необходимости,
 а не хранят всю последовательность сразу. Они особенно полезны при работе с большими объёмами данных или бесконечными последовательностями.
