### Ключевые фрагменты кода
```javascript
import { useState } from 'react';

/* =========================
   ЗАДАНИЕ 1
   Обработка массива пользователей
========================= */

const processUsers = (users) => {
    const total = users.length;

    const averageAge = total === 0
        ? 0
        : users.reduce((sum, u) => sum + u.age, 0) / total;

    const usersByCity = users.reduce((acc, u) => {
        acc[u.city] = (acc[u.city] || 0) + 1;
        return acc;
    }, {});

    const activeEmails = users
        .filter(u => u.active)
        .map(u => u.email);

    return {
        averageAge,
        usersByCity,
        activeEmails
    };
};


/* =========================
   ЗАДАНИЕ 2
   Кастомный хук useForm
========================= */

const useForm = (initialValues) => {
    const [values, setValues] = useState(initialValues);
    const [errors, setErrors] = useState({});

    const validate = () => {
        const newErrors = {};
        Object.keys(values).forEach(key => {
            if (!values[key]) {
                newErrors[key] = 'Обязательное поле';
            }
        });
        setErrors(newErrors);
        return Object.keys(newErrors).length === 0;
    };

    const handleChange = (e) => {
        const { name, value } = e.target;
        setValues(prev => ({ ...prev, [name]: value }));
    };

    const handleSubmit = (callback) => (e) => {
        e.preventDefault();
        if (validate()) {
            callback(values);
        }
    };

    return {
        values,
        errors,
        handleChange,
        handleSubmit
    };
};


/* =========================
   ЗАДАНИЕ 3
   Функция debounce
========================= */

const debounce = (func, delay) => {
    let timeout;

    return (...args) => {
        clearTimeout(timeout);
        timeout = setTimeout(() => {
            func(...args);
        }, delay);
    };
};
```

**Контрольные вопросы**
1.  Какие методы массивов в JavaScript относятся к функциям высшего порядка?
Функции высшего порядка для массивов — это методы массивов, которые принимают функции в качестве аргументов или возвращают функции. 
К ним относятся map, filter, reduce, forEach, some, every, find и sort.

2.  Как работают замыкания и для чего они используются в React?
Замыкания работают за счёт сохранения доступа функции к переменным из внешней области видимости. 
В React они используются для хранения и передачи состояния и пропсов в обработчиках событий, эффектах и хуках без необходимости повторного объявления данных.

3.  Что такое иммутабельность и почему она важна в React?
Иммутабельность — это принцип, при котором данные не изменяются напрямую, а вместо этого создаются их копии с изменениями. 
В React это важно для корректного определения изменений состояния и эффективной работы механизма повторного рендеринга.

4.  Как useMemo и useCallback помогают оптимизировать производительность?
useMemo и useCallback оптимизируют производительность за счёт мемоизации значений и функций. 
Они предотвращают повторные вычисления и пересоздание функций между рендерами, если их зависимости не изменились.

5.  Какие преимущества дают функциональные компоненты перед классовыми?
Функциональные компоненты имеют более простой и читаемый синтаксис, лучше поддерживают
повторное использование логики через хуки, легче тестируются и оптимизируются, а также являются современным и рекомендуемым подходом в React.
